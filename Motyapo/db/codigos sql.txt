CREATE DATABASE IF NOT EXISTS `motyapo`;
USE `motyapo`;

CREATE TABLE user (
  id_user int  AUTO_INCREMENT PRIMARY KEY ,
  nome_user varchar(200),
  email_user varchar(200),
  senha_user varchar(100));


CREATE TABLE adm (
  id_adm int  AUTO_INCREMENT PRIMARY KEY ,
  nome_adm varchar(200),
  email_adm varchar(200),
  senha_adm varchar(100),


CREATE table postagens(
    id int AUTO_INCREMENT PRIMARY KEY,
    titulo varchar(200),
    descricao text,
    data varchar(10),
    image` varchar(500) ,
    id_adm int ,
foreign key (id_adm) references supermercado(id_adm));
